def resonator_2nd_order(signal, fi, bw, sr):
    """Filtro IIR resonador: H_i(z) = 1 / (1 - 2r·cos(ω₀)·z⁻¹ + r²·z⁻²)"""
    r   = np.exp(-np.pi * bw / sr)
    w0  = 2 * np.pi * fi / sr
    a1  = -2 * r * np.cos(w0)
    a2  = r ** 2
    out = np.zeros_like(signal)
    for n in range(2, len(signal)):
        out[n] = signal[n] - a1 * out[n-1] - a2 * out[n-2]
    return out

Q1 · Caída espectral −12dB/oct

¿Por qué el espectro de la fuente glotal (voiced) tiene una caída de −12 dB/octava y no −6 dB/octava como una onda cuadrada?

Ver respuesta

La forma triangular/sierra del pulso glotal (modelo Rosenberg) contiene armónicos que decaen como 1/n² en amplitud, equivalente a −12 dB/oct. Un tren de impulsos puro (Dirac) decaería 0 dB/oct (todos los armónicos igual amplitud), una onda cuadrada −6 dB/oct (armónicos 1/n). La forma exacta del pulso glotal, con fase de cierre y apertura, produce la caída cuadrática característica de la voz humana.

def levinson_durbin(r, order):
    """
    Algoritmo iterativo O(p²) para resolver Rp·a = rp.
    Explota la estructura de Toeplitz de la matriz de autocorrelacion.
    """
    a = np.zeros(order + 1); a[0] = 1.0; E = r[0]
    for k in range(1, order + 1):
        lambda_k = sum(a[i] * r[k - i] for i in range(k)) / E
        a_new = a.copy()
        for i in range(1, k):
            a_new[i] = a[i] - lambda_k * a[k - i]
        a_new[k] = -lambda_k
        a = a_new; E = E * (1 - lambda_k ** 2)
    return a[1:], E

def lpc_analysis(frame, order=12):
    window   = np.hamming(len(frame))
    windowed = frame * window
    autocorr = np.array([np.dot(windowed, np.roll(windowed, lag))
                         for lag in range(order + 1)])
    autocorr /= autocorr[0]
    a, error = levinson_durbin(autocorr, order)
    return a, np.sqrt(error)

Q2 · Levinson-Durbin O(p²)

¿Por qué el algoritmo de Levinson-Durbin es O(p²) en lugar de O(p³) como la inversión de una matriz general?

Ver respuesta

Explota la estructura de Toeplitz de la matriz de autocorrelación: en cada iteración k, los nuevos coeficientes se calculan a partir de los anteriores usando el coeficiente de reflexión (PARCOR), sin recalcular toda la matriz. En lugar de resolver el sistema completo en cada paso, se actualiza incrementalmente de orden k-1 a k. La clave es que R[i,j] = R[|i-j|] (solo depende de la diferencia) — propiedad que el algoritmo aprovecha recursivamente.

def compute_cepstrum(signal_frame, n_fft=512):
    X        = np.fft.rfft(signal_frame, n=n_fft)
    log_mag  = np.log(np.abs(X) + 1e-10)
    cepstrum = np.fft.irfft(log_mag)
    return cepstrum

def lifter_low(cepstrum, cutoff_quefrency=30):
    """Paso-bajo: retiene tracto vocal (formantes)."""
    liftered = cepstrum.copy()
    liftered[cutoff_quefrency:-cutoff_quefrency] = 0
    return liftered

def cepstral_envelope(signal_frame, n_fft=512, cutoff=30):
    cep     = compute_cepstrum(signal_frame, n_fft)
    low_cep = lifter_low(cep, cutoff)
    return np.exp(np.fft.rfft(low_cep).real)

Q3 · Pico en quefrency

En el cepstrum de una vocal sonora (voiced) con F0=200Hz a sr=16kHz, ¿en qué muestra de quefrency aparece el pico de pitch y qué representa?

Ver respuesta

El pico aparece en la muestra 80: T0 = sr/F0 = 16000/200 = 80 muestras = 5ms de quefrency. Representa el período del pitch: la periodicidad del tren de pulsos glotales en el dominio cepstral aparece como un pico en la quefrencia igual al período T0=1/F0, exactamente igual que la periodicidad aparece como pico en la autocorrelación en el dominio temporal.

MDF Pitch Tracker con early termination

def magnitude_difference_function(frame, lag_min=20, lag_max=150):
    """MDF: mínimo en lag=T0 (señal se repite → |s[n]-s[n-lag]| ≈ 0)."""
    N       = len(frame)
    mdf     = np.zeros(lag_max - lag_min)
    min_mdf = np.inf

    for idx, lag in enumerate(range(lag_min, lag_max)):
        val = 0.0
        for n in range(N - lag):
            val += abs(frame[n] - frame[n + lag])
            if val > min_mdf * 1.5:  # early termination
                val = np.inf; break
        mdf[idx] = val
        if val < min_mdf: min_mdf = val

    best_lag = lag_min + np.argmin(mdf)
    return best_lag, mdf

def pitch_track(y, sr, frame_size=512, hop=128, lag_min=20, lag_max=150):
    f0_track = []
    for i in range(0, len(y) - frame_size, hop):
        frame  = y[i:i + frame_size]
        zcr    = np.sum(np.abs(np.diff(np.sign(frame)))) / (2 * frame_size)
        energy = np.mean(frame ** 2)
        is_v   = (zcr < 0.15) and (energy > 0.001)
        if is_v:
            lag, _ = magnitude_difference_function(frame, lag_min, lag_max)
            f0_track.append(sr / lag)
        else:
            f0_track.append(0.0)
    return np.array(f0_track)

Q4 · Mínimo de MDF

¿Por qué la MDF tiene un mínimo (no un máximo) en el lag igual al período de pitch, mientras que la ACF tiene un máximo?

Ver respuesta

Cuando el retardo l = T0 (período), las muestras s[n] y s[n-l] son casi idénticas porque la señal se repite periódicamente, por lo que |s[n]−s[n−l]| ≈ 0 y la suma MDF(l) es mínima. La ACF es el producto s[n]·s[n-l], que es máximo cuando las señales son idénticas (producto de iguales = máximo). Son complementarias: ambas detectan periodicidad pero con criterio opuesto.

def compute_mfcc_voice(y, sr=16000, n_mfcc=13, n_mels=40,
                        n_fft=512, hop_length=160, pre_emph=0.97):
    # Pre-enfasis: compensa -12dB/oct de laringe (crucial en voz, no en musica)
    y_pe = np.append(y[0], y[1:] - pre_emph * y[:-1])

    # STFT
    D   = librosa.stft(y_pe, n_fft=n_fft, hop_length=hop_length, window='hamming')
    mag = np.abs(D)

    # Banco Mel: rango 80-7600 Hz (voz: F0 minimo a fricativas altas)
    mel_basis = librosa.filters.mel(sr=sr, n_fft=n_fft, n_mels=n_mels,
                                     fmin=80, fmax=7600)
    log_mel   = np.log(mel_basis @ mag + 1e-8)

    # DCT-II: decorrelacion (aprox. Karhunen-Loeve)
    from scipy.fft import dct
    mfcc   = dct(log_mel, type=2, axis=0, norm='ortho')[:n_mfcc, :]

    # Delta y delta-delta: captura dinamica temporal del habla
    delta1 = librosa.feature.delta(mfcc, order=1)
    delta2 = librosa.feature.delta(mfcc, order=2)
    return np.vstack([mfcc, delta1, delta2])  # (39, T) si n_mfcc=13

def speaker_embedding(y, sr=16000):
    """Vector de identidad del locutor (156-d)."""
    mfcc_full = compute_mfcc_voice(y, sr)
    return np.concatenate([
        mfcc_full.mean(axis=1),
        mfcc_full.std(axis=1),
        np.percentile(mfcc_full, 5,  axis=1),
        np.percentile(mfcc_full, 95, axis=1),
    ])

Q5 · Rango Mel 80–7600Hz

¿Por qué el banco Mel para voz se limita a 80–7600 Hz en lugar de 20–8000 Hz para audio general?

Ver respuesta

La voz humana contiene la energía fonética relevante entre 80 Hz (F0 mínimo del habla) y 7600 Hz (fricativas altas /s/, /f/). Los extremos son ruido de fondo o armónicos inaudibles que reducen la discriminabilidad del modelo sin aportar información fonética. El filtrado de rango mejora el SNR efectivo de los coeficientes y el rendimiento en ASR y speaker recognition.

M6 · Modelado Físico
Digital Waveguide Synthesis — Kelly-Lochbaum
Modelado físico del tracto como guía de onda digital: dispersión bidireccional en uniones KL
Coeficiente reflexión: k = (A₂ − A₁) / (A₂ + A₁) P⁺₂ = (1+k)·P⁺₁ + (−k)·P⁻₂ [transmisión hacia boca] P⁻₁ = k·P⁺₁ + (1−k)·P⁻₂ [reflexión hacia laringe]

Digital Waveguide Vocal Tract — Kelly-Lochbaum

class KellyLochbaumTract:
    """N tubos cilindricos concatenados. Cada uno = delay de 1 muestra."""
    def __init__(self, areas, sr=16000):
        self.N  = len(areas)
        self.k  = np.array([
            (areas[i+1] - areas[i]) / (areas[i+1] + areas[i] + 1e-10)
            for i in range(self.N - 1)
        ])
        self.pf = np.zeros(self.N)  # onda hacia la boca (forward)
        self.pb = np.zeros(self.N)  # onda hacia laringe (backward)

    def process_sample(self, glottal_input):
        pf_new = self.pf.copy(); pb_new = self.pb.copy()
        pf_new[0] = glottal_input + self.pb[0] * (-1.0)  # reflexion en laringe

        for i in range(self.N - 1):
            k           = self.k[i]
            pf_new[i+1] = (1 + k) * self.pf[i] + (-k) * self.pb[i+1]
            pb_new[i]   = k * self.pf[i] + (1 - k) * self.pb[i+1]

        output  = pf_new[-1]  # presion en la boca
        self.pf = np.roll(pf_new, 1); self.pb = np.roll(pb_new, -1)
        return output

Q6 · Reflexión Kelly-Lochbaum

¿Qué sucede en la unión Kelly-Lochbaum cuando A₁ = A₂ (dos tubos del mismo área)? Justifica matemáticamente.

Ver respuesta

k = (A₂ − A₁) / (A₂ + A₁) = 0/2A₁ = 0: sin reflexión. La onda se transmite íntegra sin perturbación, como una línea de transmisión continua sin discontinuidad de impedancia acústica. P⁺₂ = (1+0)·P⁺₁ + 0 = P⁺₁ y P⁻₁ = 0 + (1-0)·P⁻₂ = P⁻₂ — transmisión perfecta en ambas direcciones.

M7 · WaveNet
WaveNet — Convoluciones Causales Dilatadas
Síntesis neural de voz a nivel de muestra: campo receptivo exponencial sin RNN
Dilation d: Conv(x)[n] = Σₖ w[k] · x[n − d·k] z = tanh(Wf * x) ⊙ σ(Wg * x) [Gated Activation] Residual: h = z + x Capas [1,2,4,8,...,512]: campo receptivo = 1024 muestras ≈ 23ms @ 44.1kHz

WaveNet simplificado en PyTorch

class CausalConv1d(nn.Module):
    def __init__(self, in_ch, out_ch, kernel_size, dilation=1):
        super().__init__()
        self.padding = (kernel_size - 1) * dilation
        self.conv = nn.Conv1d(in_ch, out_ch, kernel_size, dilation=dilation, padding=0)
    def forward(self, x):
        return self.conv(F.pad(x, (self.padding, 0)))  # padding causal: solo al pasado

class WaveNetLayer(nn.Module):
    def __init__(self, channels=32, kernel_size=2, dilation=1):
        super().__init__()
        self.filter_conv = CausalConv1d(channels, channels, kernel_size, dilation)
        self.gate_conv   = CausalConv1d(channels, channels, kernel_size, dilation)
        self.res_conv    = nn.Conv1d(channels, channels, 1)
    def forward(self, x):
        z = torch.tanh(self.filter_conv(x)) * torch.sigmoid(self.gate_conv(x))
        return self.res_conv(z) + x, z

# µ-law quantization: 256 niveles perceptualmente equivalentes a ~16-bit lineal
def mu_law_encode(x, mu=255):
    x_clipped = x.clip(-1, 1)
    encoded   = np.sign(x_clipped) * np.log1p(mu * np.abs(x_clipped)) / np.log1p(mu)
    return ((encoded + 1) / 2 * mu + 0.5).astype(np.int32)

def mu_law_decode(encoded, mu=255):
    x = 2 * encoded / mu - 1
    return np.sign(x) * (np.exp(np.abs(x) * np.log1p(mu)) - 1) / mu

Q7 · Inferencia lenta de WaveNet

WaveNet es autoregresivo: genera muestra a muestra. ¿Por qué esto lo hace extremadamente lento en inferencia a pesar de ser convoluciones (que en training son paralelas)?

Ver respuesta

Porque no puede paralelizar la generación: cada muestra depende de todas las anteriores (causalidad estricta). T muestras requieren T pasadas forward secuenciales — imposible batching temporal. En training sí se paraleliza mediante teacher forcing (la señal real se da como entrada), pero en inferencia el output de t es el input de t+1. Para 1 segundo a 16kHz = 16000 pasos secuenciales.

DDSP — Procesamiento de Señal Diferenciable
Síntesis físicamente informada: osciladores y filtros como nodos diferenciables

DDSP Synthesizer + Multi-Scale Spectral Loss

def additive_synth(f0, amplitudes, sr=16000, n_samples=16000):
    """Sintesis aditiva diferenciable."""
    B, T, n_harmonics = amplitudes.shape
    harmonic_freqs = f0.unsqueeze(-1) * torch.arange(1, n_harmonics+1, device=f0.device)
    phases  = torch.cumsum(2 * np.pi * harmonic_freqs / sr, dim=1)
    return (amplitudes * torch.sin(phases)).sum(dim=-1)

def filtered_noise_synth(magnitudes, n_samples=16000):
    """Ruido blanco filtrado en frecuencia (fase lineal)."""
    B       = magnitudes.shape[0]
    noise   = torch.randn(B, n_samples, device=magnitudes.device)
    noise_f = torch.fft.rfft(noise, n=magnitudes.shape[1] * 2 - 2)
    return torch.fft.irfft(noise_f * magnitudes[:, :noise_f.shape[1]], n=n_samples)

class MultiScaleSpectralLoss(nn.Module):
    def __init__(self, fft_sizes=(512, 1024, 2048)):
        super().__init__(); self.fft_sizes = fft_sizes
    def forward(self, x_real, x_pred):
        loss = 0.0
        for n_fft in self.fft_sizes:
            hop  = n_fft // 4
            S_r  = torch.abs(torch.stft(x_real, n_fft, hop, return_complex=True))
            S_p  = torch.abs(torch.stft(x_pred, n_fft, hop, return_complex=True))
            # Convergencia espectral + log-magnitud L1
            sc   = torch.norm(S_r - S_p, p='fro') / (torch.norm(S_r, p='fro') + 1e-8)
            mag  = (torch.log(S_r + 1e-8) - torch.log(S_p + 1e-8)).abs().mean()
            loss += sc + mag
        return loss / len(self.fft_sizes)

Q8 · Pérdida espectral multi-resolución

¿Por qué DDSP usa pérdida multi-resolución (FFT 512, 1024, 2048) en lugar de MSE en el dominio del tiempo?

Ver respuesta

El MSE en tiempo es muy sensible a desajustes de fase: dos sinusoides idénticas desfasadas en π tienen MSE máximo aunque suenen igual perceptualmente. La pérdida espectral (magnitudes STFT) es invariante a fase y produce gradientes más informativos sobre el contenido tonal. La multi-resolución es necesaria porque FFTs pequeñas (512) capturan transitorios (buena resolución temporal) y FFTs grandes (2048) capturan estructura tonal (buena resolución frecuencial) — cubriendo el trade-off de Heisenberg.

LPCNet GRU Architecture

class LPCNetGRU(nn.Module):
    def __init__(self, feature_dim=20, hidden_size=128, n_outputs=256):
        super().__init__()
        self.feature_proj = nn.Linear(feature_dim, hidden_size)
        self.gru_a = nn.GRU(hidden_size + 1, hidden_size, batch_first=True)
        self.gru_b = nn.GRU(hidden_size, hidden_size, batch_first=True)
        self.output = nn.Linear(hidden_size, n_outputs)

    def forward(self, features, excitation, h_a=None, h_b=None):
        """
        features:   (B, T_frame, 20)  — features cada 10ms (LPC + pitch)
        excitation: (B, T_sample, 1)  — muestra de excitacion anterior
        """
        feat_proj = torch.relu(self.feature_proj(features))
        feat_rep  = feat_proj.repeat_interleave(160, dim=1)  # interpolar frame→sample
        gru_input = torch.cat([feat_rep[:, :excitation.shape[1], :], excitation], dim=-1)
        out_a, h_a = self.gru_a(gru_input, h_a)
        out_b, h_b = self.gru_b(out_a, h_b)
        return self.output(out_b), h_a, h_b  # logits de distribucion µ-law

# LPCNet real: ~1M params, 15 MFLOPS por muestra en C optimizado (tiempo real en CPU)
# WaveNet equivalente: miles de canales con dilated convs, necesita GPU
# Clave: GRU solo modela residuo (señal simple, casi blanca) — no la voz completa

Q9 · LPCNet vs WaveNet eficiencia

¿Por qué LPCNet puede ejecutarse en tiempo real en CPU (15 MFLOPS) mientras WaveNet necesita GPU, manteniendo calidad comparable?

Ver respuesta

La GRU de LPCNet solo necesita modelar el residuo de LPC (señal simple, casi blanca — el predictor lineal ya eliminó la estructura periódica del habla). La distribución del residuo es mucho más simple que la distribución de la señal de voz completa. Esto permite una GRU pequeña de 128 unidades vs WaveNet que necesita miles de canales para modelar la distribución completa sample-a-sample de la forma de onda.

def estimate_bitrate(order=10, frame_ms=20):
    frames_per_sec = 1000 / frame_ms  # 50 frames/s
    bits_voiced    = 1
    bits_f0        = 7   # 128 niveles → 60-500 Hz
    bits_lpc       = order * 6  # 6 bits por coeficiente
    bits_gain      = 5   # 32 niveles
    bits_per_frame = bits_voiced + bits_f0 + bits_lpc + bits_gain
    bitrate        = bits_per_frame * frames_per_sec
    print(f"Vocoder parametrico: {bitrate:.0f} bps")
    return bitrate

estimate_bitrate(order=10, frame_ms=20)  # ~750 bps, cerca de MELPe

Q10 · MELPe 600bps vs PCM 64kbps

MELPe a 600bps es inteligible pero suena sintético. ¿Qué información exacta se pierde respecto a PCM a 64kbps?

Ver respuesta

MELPe transmite solo los parámetros: F0, periodicidad, 10 coefs LPC y bandas de ruido. La voz se reconstituye sintetizando excitación con pulsos/ruido ideales. PCM preserva cada muestra de la forma de onda real, incluyendo todas las irregularidades glotales, la microestructura del residuo de predicción, las variaciones ciclo-a-ciclo del pitch y los transitorios de consonantes — todo esto da "naturalidad" que el modelo paramétrico no puede capturar con solo F0 y LPC.

class ResBlock(nn.Module):
    def __init__(self, channels, kernel_size=3, dilations=(1,3,5)):
        super().__init__()

            nn.utils.weight_norm(nn.Conv1d(

Q11 · Discriminadores primos en HiFi-GAN

HiFi-GAN usa discriminadores en períodos primos [2,3,5,7,11]. ¿Por qué específicamente primos y no cualquier selección de períodos?

Ver respuesta

Los períodos primos son coprimos entre sí (MCD=1 para cualquier par). Esto garantiza que cada discriminador captura una estructura periódica diferente sin solapamiento entre sus patrones de análisis. Por ejemplo, un discriminador con período 6 cubriría parcialmente los de períodos 2 y 3. Con primos, se maximiza la diversidad de perspectivas sobre la señal, cubriendo un amplio espectro de periodicidades armónicas sin redundancia.

Q12 · VAE separación contenido/timbre

¿Por qué el VAE para voice conversion separa el espacio latente en contenido fonético y estilo/timbre? ¿Qué permite esta separación que no es posible sin ella?

Ver respuesta

Para poder recombinar: el encoder comprime por separado el contenido fonético (qué se dice — fonemas, prosodia, invariante al locutor) y el estilo tímbrico (cómo suena — identidad del locutor). El decoder puede entonces generar la voz del locutor target combinando el contenido de A con el timbre de B. Sin esta separación, el espacio latente mezcla ambas dimensiones y es imposible manipular una sin alterar la otra.

def apply_itd_ild(signal, itd_us=300, ild_db=6, sr=44100):
    """Fuente a la derecha: oido izquierdo llega tarde y mas debil."""
    itd_samples = int(itd_us * 1e-6 * sr)
    right = signal.copy()
    left  = np.zeros(len(signal) + itd_samples)
    left[itd_samples:] = signal
    left  = left[:len(signal)]
    left *= 10 ** (-ild_db / 20)  # ILD: el oido mas lejano recibe menos nivel
    return np.column_stack([left, right])

def haas_demo(voice, sr=44100, delay_ms=20):
    """
    delay_ms < 40ms: fusion → percepcion de un solo evento
    delay_ms 40-80ms: eco perturbador que degrada inteligibilidad
    delay_ms > 80ms: eco claramente separado
    """
    delay_samples = int(delay_ms * 0.001 * sr)
    reflection    = np.concatenate([np.zeros(delay_samples), voice])[:len(voice)] * 0.7
    return voice + reflection

# ITD 300µs @ 44100Hz = 13 muestras de retardo
stereo = apply_itd_ild(voice, itd_us=300, ild_db=6, sr=44100)

Q13 · Efecto Haas 40–80ms

El efecto Haas funciona con retardos <40ms (fusión). ¿Qué ocurre exactamente con retardos de 40–80ms? ¿Por qué es la zona más problemática?

Ver respuesta

Los dos sonidos se perciben como eventos separados (la imagen de fusión se rompe), pero la reflexión todavía no es claramente un eco distinto: es la zona de "eco perturbador" o "haas breakout" que degrada la inteligibilidad sin ser percibido como eco natural. El cerebro ya no puede fusionar los dos eventos pero tampoco los separa cognitivamente como señal directa + reflexión ambiental — produce confusión perceptual y reduce la claridad de la voz.

Q1 · Orden LPC y sobreajuste

Si aumentas el orden LPC de p=10 a p=20 en una señal de voz a 8kHz, ¿qué sucede con la calidad de síntesis y por qué existe un orden óptimo?

Ver respuesta

A 8kHz con p=20, el filtro puede empezar a modelar la excitación glotal en lugar de solo el tracto, causando sobreajuste: el residuo de predicción deja de ser "casi blanco" y contiene estructura del tracto — la síntesis degrada porque el filtro de síntesis 1/A(z) ya no representa puramente el tracto. La regla: p ≈ sr_kHz × 1.5 + 4 para voz (2 polos por formante). A 8kHz: p ≈ 12–16 máximo.

Q2 · Cepstrum F0 y quefrency

Si quieres estimar el pitch via cepstrum y la señal tiene F0=200Hz a sr=16kHz, ¿en qué muestra exacta de quefrency buscas el pico y cómo lo calculas?

Ver respuesta

Muestra 80: T0 = sr/F0 = 16000/200 = 80 muestras = 5ms de quefrency. El procedimiento: calcular cepstrum, aplicar lifter de paso-alto (cortar quefrencias bajas del tracto, retener altas del pitch), buscar el pico más alto en el rango [sr/F0_max, sr/F0_min] = [32, 267] muestras para voz (F0 entre 60–500Hz). El pico en muestra 80 corresponde exactamente a la periodicidad de la señal.

Q3 · WaveNet vs LPCNet en inferencia

¿Por qué WaveNet necesita GPU para inferencia en tiempo real pero LPCNet puede correr en CPU de bajo coste, siendo ambos modelos autoregresivos?

Ver respuesta

WaveNet necesita miles de canales con dilated convolutions para modelar la distribución completa de la forma de onda sample-a-sample: cada paso forward involucra grandes operaciones de convolución en paralelo que se benefician de GPU. LPCNet solo modela el residuo de LPC con una GRU de 128 unidades (el predictor lineal ya factorizó la mayor parte de la estructura): 15 MFLOPS por muestra vs cientos de MFLOPS de WaveNet. La parte más costosa (la predicción lineal de LPC) es O(p) y se hace en C nativo.

Q4 · DDSP Multi-Scale Loss

¿Por qué DDSP usa pérdida multi-resolución espectral en lugar de MSE en el dominio temporal? Explica el principio de incertidumbre de Heisenberg en este contexto.

Ver respuesta

MSE en tiempo es sensible a desajustes de fase: dos sinusoides idénticas desfasadas en π tienen MSE máximo aunque suenen igual — gradientes no informativos. La pérdida espectral (magnitudes) es invariante a fase. La multi-resolución aborda el principio de Heisenberg tiempo-frecuencia: no puedes tener buena resolución temporal y frecuencial simultáneamente con un solo tamaño de FFT. FFT 512 = buena resolución temporal (transitorios), FFT 2048 = buena resolución frecuencial (tonos sostenidos). La combinación cubre el espacio completo.

Q5 · HiFi-GAN upsample ratio

HiFi-GAN usa upsample rates [8,8,2,2] para un mel-spectrograma extraído con hop_length=256 a 22050Hz. ¿Qué ratio total produce y por qué debe coincidir con el hop_length?

Ver respuesta

8×8×2×2 = 256× total. Debe coincidir con el hop_length porque el mel-spectrograma tiene 1 frame por cada hop_length muestras del audio original. Si el hop_length es 256, cada frame mel representa 256 muestras de audio. El generador debe producir exactamente 256 muestras por frame mel para reconstruir una señal de longitud T_audio = T_frames × 256 muestras. Con ratio 256 y 100 frames mel: 25600 muestras = 1.16s a 22050Hz.
