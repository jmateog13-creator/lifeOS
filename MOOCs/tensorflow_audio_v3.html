STFT Differentiable en tf.signal

import tensorflow as tf
import numpy as np

def compute_stft_spectrogram(waveform: tf.Tensor,
                              frame_length: int = 1024,
                              frame_step: int = 256,
                              fft_length: int = 1024) -> tf.Tensor:
    """
    STFT completamente dentro del grafo TF — diferenciable.
    Parámetros:
        waveform: [batch, samples] float32
    Returns:
        magnitude: [batch, frames, fft_length//2+1] float32
    """
    # Análisis STFT: ventana Hann
    stfts = tf.signal.stft(
        waveform,
        frame_length=frame_length,
        frame_step=frame_step,
        fft_length=fft_length,
        window_fn=tf.signal.hann_window,
        pad_end=True
    )
    # stfts: complex64 [batch, frames, fft_length//2+1]
    magnitude = tf.abs(stfts)
    power = tf.square(magnitude)  # |X|²
    return power

def compute_log_mel(waveform: tf.Tensor,
                    sample_rate: int = 16000,
                    num_mel_bins: int = 128,
                    lower_edge_hz: float = 80.0,
                    upper_edge_hz: float = 7600.0) -> tf.Tensor:
    """Log-Mel spectrogram: feature estándar para clasificación."""
    power = compute_stft_spectrogram(waveform)

    # Mel filterbank matrix: [fft_length//2+1, num_mel_bins]
    num_spectrogram_bins = power.shape[-1]
    mel_matrix = tf.signal.linear_to_mel_weight_matrix(
        num_mel_bins, num_spectrogram_bins, sample_rate,
        lower_edge_hz, upper_edge_hz
    )
    mel_power = tf.tensordot(power, mel_matrix, axes=1)  # [..., frames, mel]

    log_mel = tf.math.log(mel_power + 1e-6)  # +eps para estabilidad numérica
    return log_mel  # [batch, frames, num_mel_bins]

RaggedTensor para Audio de Longitud Variable

# El problema: audios de diferente duración → no se pueden apilar en tensor denso
# Solución: tf.RaggedTensor — flat_values + row_splits

def process_variable_length_audio(audio_files: list[str]) -> tf.RaggedTensor:
    """
    Procesa archivos de audio de longitud variable sin padding.
    Estructura: flat_values (todas las muestras concatenadas) + row_splits
    """
    waveforms = []
    for path in audio_files:
        audio_binary = tf.io.read_file(path)
        waveform, sr = tf.audio.decode_wav(audio_binary)  # [samples, channels]
        waveform = tf.squeeze(waveform, axis=-1)           # [samples]
        waveforms.append(waveform)

    # Crear RaggedTensor automáticamente desde lista de tensores
    ragged = tf.ragged.stack(waveforms)
    # ragged.flat_values: todas las muestras concatenadas
    # ragged.row_splits: índices de inicio/fin de cada audio

    # Extraer STFT con map (cada fila independiente)
    def extract_features(wav):
        stft = tf.signal.stft(wav, frame_length=1024, frame_step=256)
        return tf.abs(stft)

    features = tf.map_fn(extract_features, ragged, fn_output_signature=tf.RaggedTensorSpec(
        shape=[None, 513], dtype=tf.float32
    ))
    return features
M2 · Pérdida
Multi-Resolution STFT Loss
L_sc + L_mag sobre M tamaños de FFT — generación de audio de alta calidad
L_MR-STFT = (1/M) · Σₘ [L_sc(m) + L_mag(m)]
L_sc(m) = ||STFT(x)| − |STFT(x̂)|| / ||STFT(x)|| (spectral convergence)
L_mag(m) = ||log|STFT(x)| − log|STFT(x̂)||₁ (log-magnitude)

Implementación MR-STFT Loss

import tensorflow as tf

class MultiResolutionSTFTLoss(tf.keras.losses.Loss):
    """
    Multi-Scale STFT Loss — usado en HiFi-GAN, DDSP, EnCodec.
    Múltiples resoluciones tiempo-frecuencia capturan:
    - FFT pequeña (256): buena resolución temporal, pobre freq.
    - FFT grande (2048): buena resolución freq., pobre temporal.
    """
    def __init__(self, fft_sizes=(256, 512, 1024, 2048),
                       hop_sizes=(64, 128, 256, 512),
                       win_sizes=(256, 512, 1024, 2048)):
        super().__init__()
        assert len(fft_sizes) == len(hop_sizes) == len(win_sizes)
        self.configs = list(zip(fft_sizes, hop_sizes, win_sizes))

    def _stft_magnitude(self, wav, fft_size, hop_size, win_size):
        stft = tf.signal.stft(wav, frame_length=win_size,
                               frame_step=hop_size, fft_length=fft_size,
                               window_fn=tf.signal.hann_window)
        return tf.abs(stft)

    def call(self, y_true: tf.Tensor, y_pred: tf.Tensor) -> tf.Tensor:
        total_loss = 0.0
        for fft_size, hop_size, win_size in self.configs:
            mag_true = self._stft_magnitude(y_true, fft_size, hop_size, win_size)
            mag_pred = self._stft_magnitude(y_pred, fft_size, hop_size, win_size)

            # Spectral Convergence Loss
            sc_loss = tf.norm(mag_true - mag_pred, ord='fro', axis=[-2,-1])
            sc_loss /= (tf.norm(mag_true, ord='fro', axis=[-2,-1]) + 1e-7)

            # Log-Magnitude Loss (L1)
            log_mag_true = tf.math.log(mag_true + 1e-7)
            log_mag_pred = tf.math.log(mag_pred + 1e-7)
            mag_loss = tf.reduce_mean(tf.abs(log_mag_true - log_mag_pred))

            total_loss += tf.reduce_mean(sc_loss) + mag_loss

        return total_loss / len(self.configs)

AST — Audio Spectrogram Transformer

import tensorflow as tf

class AudioPatchEmbedding(tf.keras.layers.Layer):
    """Divide el espectrograma en patches 16×16 y proyecta a d_model."""
    def __init__(self, patch_size=16, d_model=768):
        super().__init__()
        self.patch_size = patch_size
        self.proj = tf.keras.layers.Dense(d_model)

    def call(self, x: tf.Tensor) -> tf.Tensor:
        # x: [batch, time_frames, freq_bins, 1]
        B, T, F, C = x.shape
        ps = self.patch_size
        # Reshape a patches
        patches = tf.image.extract_patches(
            x, sizes=[1,ps,ps,1], strides=[1,ps,ps,1], rates=[1,1,1,1], padding='VALID'
        )
        # patches: [B, T//ps, F//ps, ps*ps*C]
        patches = tf.reshape(patches, [B, -1, ps*ps*C])
        return self.proj(patches)  # [B, num_patches, d_model]

class AudioSpectrogramTransformer(tf.keras.Model):
    """AST: ViT aplicado a espectrogramas de audio."""
    def __init__(self, num_classes=527, d_model=768, num_heads=12, num_layers=12):
        super().__init__()
        self.patch_embed = AudioPatchEmbedding(16, d_model)
        self.cls_token = self.add_weight('cls', [1,1,d_model], initializer='zeros')
        self.pos_embed = self.add_weight('pos', [1, 513, d_model], initializer='zeros')
        self.transformer_blocks = [
            tf.keras.layers.MultiHeadAttention(num_heads, d_model//num_heads)
            for _ in range(num_layers)
        ]
        self.layer_norms = [tf.keras.layers.LayerNormalization() for _ in range(num_layers)]
        self.head = tf.keras.layers.Dense(num_classes, activation='sigmoid')  # AudioSet multi-label

    def call(self, x, training=False):
        B = tf.shape(x)[0]
        x = self.patch_embed(x)                    # [B, N, d_model]
        cls = tf.broadcast_to(self.cls_token, [B, 1, x.shape[-1]])
        x = tf.concat([cls, x], axis=1)            # prepend [CLS]
        x += self.pos_embed[:, :tf.shape(x)[1]]
        for attn, norm in zip(self.transformer_blocks, self.layer_norms):
            residual = x
            x = norm(x)
            x = attn(x, x) + residual              # self-attention + skip
        return self.head(x[:, 0])                  # [CLS] token → logits

import tensorflow as tf
import numpy as np

AUTOTUNE = tf.data.AUTOTUNE

def load_and_decode_wav(filepath: tf.Tensor, label: tf.Tensor):
    """Carga un archivo WAV y retorna waveform + label."""
    audio_binary = tf.io.read_file(filepath)
    waveform, sr = tf.audio.decode_wav(audio_binary, desired_channels=1)
    waveform = tf.squeeze(waveform, -1)
    # Pad/truncate a longitud fija (4 seg @ 16kHz)
    target_len = 64000
    waveform = waveform[:target_len]
    waveform = tf.pad(waveform, [[0, tf.maximum(0, target_len - tf.shape(waveform)[0])]])
    return waveform, label

def audio_augment(waveform: tf.Tensor, label: tf.Tensor):
    """Augmentación en el dominio de la señal (on-the-fly)."""
    # 1. Gaussian noise
    noise = tf.random.normal(tf.shape(waveform), stddev=0.005)
    waveform = waveform + noise
    # 2. Random gain
    gain = tf.random.uniform([], 0.8, 1.2)
    waveform = waveform * gain
    # 3. Random pitch shift no disponible en tf.signal → usar scipy en preprocessing
    return tf.clip_by_value(waveform, -1.0, 1.0), label

def specaugment(log_mel: tf.Tensor, label: tf.Tensor,
                time_mask_param=40, freq_mask_param=20) -> tuple:
    """SpecAugment: máscaras temporales y de frecuencia."""
    # Time mask
    t = tf.shape(log_mel)[0]
    t0 = tf.random.uniform([], 0, t - time_mask_param, dtype=tf.int32)
    mask_t = tf.concat([tf.ones([t0]), tf.zeros([time_mask_param]),
                         tf.ones([t - t0 - time_mask_param])], axis=0)
    log_mel = log_mel * mask_t[:, tf.newaxis]
    # Freq mask
    f = tf.shape(log_mel)[1]
    f0 = tf.random.uniform([], 0, f - freq_mask_param, dtype=tf.int32)
    mask_f = tf.concat([tf.ones([f0]), tf.zeros([freq_mask_param]),
                         tf.ones([f - f0 - freq_mask_param])], axis=0)
    log_mel = log_mel * mask_f[tf.newaxis, :]
    return log_mel, label

def build_dataset(file_paths: list, labels: list, batch_size: int = 32,
                  training: bool = True) -> tf.data.Dataset:
    ds = tf.data.Dataset.from_tensor_slices((file_paths, labels))

    if training:
        ds = ds.shuffle(10000, reshuffle_each_iteration=True)

    # Carga paralela con interleave (evita bottleneck de I/O)
    ds = ds.interleave(
        lambda f, l: tf.data.Dataset.from_tensors(load_and_decode_wav(f, l)),
        cycle_length=16,
        num_parallel_calls=AUTOTUNE
    )

    if training:
        ds = ds.map(audio_augment, num_parallel_calls=AUTOTUNE)

    # Calcular Log-Mel
    ds = ds.map(lambda w, l: (tf.math.log(
        tf.tensordot(tf.square(tf.abs(tf.signal.stft(w, 1024, 256))),
                     tf.signal.linear_to_mel_weight_matrix(128, 513, 16000, 80, 7600), 1) + 1e-6
    ), l), num_parallel_calls=AUTOTUNE)

    if training:
        ds = ds.map(specaugment, num_parallel_calls=AUTOTUNE)

    ds = ds.batch(batch_size).prefetch(AUTOTUNE)
    return ds
M5 · Quantización
QAT — Quantization-Aware Training
Fake quantization, Straight-Through Estimator (STE), int8 inference
Q(x) = Δ · round(clip(x/Δ, l, u))
Δ = (x_max − x_min) / (2^b − 1) (escala de cuantización)
STE: ∂L/∂x ≈ ∂L/∂Q(x) (gradiente pasa directamente en backprop)

QAT con TensorFlow Model Optimization

import tensorflow as tf
import tensorflow_model_optimization as tfmot

# ====================================================
# QAT: entrenar con "fake quantization" nodes
# ====================================================
def apply_qat(model: tf.keras.Model) -> tf.keras.Model:
    """
    Aplica QAT: inserta FakeQuant nodes antes de cada capa cuantizable.
    Los FakeQuant simulan el error de cuantización int8 durante el entrenamiento.
    STE asegura que el gradiente fluye como si no hubiera cuantización.
    """
    quantize_model = tfmot.quantization.keras.quantize_model
    qat_model = quantize_model(model)
    return qat_model

# ====================================================
# Ejemplo manual de Straight-Through Estimator (STE)
# ====================================================
@tf.custom_gradient
def fake_quantize(x: tf.Tensor):
    """Cuantización round-to-nearest con STE en backprop."""
    # Forward: cuantizar a 8 bits
    x_min = tf.reduce_min(x)
    x_max = tf.reduce_max(x)
    scale = (x_max - x_min) / 255.0  # int8: 256 niveles
    x_q = tf.round(tf.clip_by_value(x / scale, -128, 127)) * scale

    def grad(upstream):
        # STE: gradiente de la función identidad (no de round)
        # Solo en rango de cuantización: clip implícito
        pass_through = tf.where(
            tf.logical_and(x >= x_min, x <= x_max),
            upstream, tf.zeros_like(upstream)
        )
        return pass_through

    return x_q, grad

# ====================================================
# Post-Training Quantization (PTQ) — más rápido, menor calidad
# ====================================================
def ptq_to_tflite(model: tf.keras.Model,
                   representative_dataset) -> bytes:
    """Convierte a TFLite int8 con calibración."""
    converter = tf.lite.TFLiteConverter.from_keras_model(model)
    converter.optimizations = [tf.lite.Optimize.DEFAULT]
    converter.representative_dataset = representative_dataset
    converter.target_spec.supported_ops = [tf.lite.OpsSet.TFLITE_BUILTINS_INT8]
    converter.inference_input_type = tf.int8
    converter.inference_output_type = tf.int8
    return converter.convert()

def representative_audio_dataset():
    """100 samples representativas para calibración PTQ."""
    for _ in range(100):
        sample = tf.random.normal([1, 128, 128, 1])  # log-mel
        yield [sample]

Mixed Precision en Keras

import tensorflow as tf

# Activar mixed precision globalmente
tf.keras.mixed_precision.set_global_policy('mixed_float16')

# Las capas Dense/Conv usan FP16 para las computaciones
# pero acumulan en FP32 para mantener precisión numérica

# LossScaleOptimizer previene underflow de gradientes FP16
# (FP16: rango [-65504, 65504], gradientes pequeños → 0 sin scaling)
base_optimizer = tf.keras.optimizers.Adam(1e-4)
optimizer = tf.keras.mixed_precision.LossScaleOptimizer(base_optimizer)

# Entrenar con escalado de pérdida manual (eager mode)
@tf.function
def train_step(model, x, y):
    with tf.GradientTape() as tape:
        y_pred = model(x, training=True)
        loss = compute_loss(y, y_pred)
        scaled_loss = optimizer.get_scaled_loss(loss)

    scaled_grads = tape.gradient(scaled_loss, model.trainable_variables)
    grads = optimizer.get_unscaled_gradients(scaled_grads)
    optimizer.apply_gradients(zip(grads, model.trainable_variables))
    return loss

# Beneficios en A100/V100: ~2x throughput (Tensor Cores FP16)
# Modelo ocupa ~50% memoria → batch size 2x mayor

Reglas de Mixed Precision

  • Matmul/Conv en FP16: Tensor Cores → 2-8x más rápido
  • Acumulación en FP32: Evita errores de redondeo en sumas largas
  • BatchNorm en FP32: La varianza con FP16 puede ser inestable
  • Loss Scaling: Multiplica la pérdida por 2^7–2^17, divide gradientes → evita underflow
  • Softmax en FP32: exponenciales FP16 pueden desbordarse
Cuidado: Si el loss scale baja continuamente (overflow de gradientes), el modelo tiene NaN. Verificar con optimizer.loss_scale durante entrenamiento.

interpreter.AllocateTensors();

// Input: log-mel spectrogram [1, 32, 64, 1] int8
TfLiteTensor* input = interpreter.input(0);

// Llenar con datos de micrófono (cuantizados)
void run_inference(int8_t* log_mel_data, int len) {
    memcpy(input->data.int8, log_mel_data, len);
    interpreter.Invoke();
    TfLiteTensor* output = interpreter.output(0);
    // output->data.int8 → índice de clase con max = predicción
    int predicted_class = 0;
    for (int i = 1; i < output->dims->data[1]; i++) {
        if (output->data.int8[i] > output->data.int8[predicted_class])
            predicted_class = i;
    }
}

import tensorflow as tf

class ResBlock(tf.keras.layers.Layer):
    """Bloque residual con convoluciones dilatadas."""
    def __init__(self, channels, kernel_size=3, dilations=(1,3,5)):
        super().__init__()
        self.convs = []
        for d in dilations:
            self.convs.append(tf.keras.Sequential([
                tf.keras.layers.LeakyReLU(0.1),
                tf.keras.layers.Conv1D(channels, kernel_size, padding='same', dilation_rate=d),
                tf.keras.layers.LeakyReLU(0.1),
                tf.keras.layers.Conv1D(channels, kernel_size, padding='same'),
            ]))

    def call(self, x):
        for conv in self.convs:
            x = x + conv(x)  # skip connection
        return x

class HiFiGANGenerator(tf.keras.Model):
    """
    Generador HiFi-GAN V1:
    mel spectrogram → waveform via ConvTranspose1d upsamplers
    """
    def __init__(self, upsample_rates=(8,8,2,2),
                       upsample_kernels=(16,16,4,4),
                       resblock_channels=512,
                       resblock_kernels=(3,7,11),
                       resblock_dilations=((1,3,5),(1,3,5),(1,3,5))):
        super().__init__()
        self.pre_conv = tf.keras.layers.Conv1D(resblock_channels, 7, padding='same')

        self.ups = []
        self.resblocks = []
        ch = resblock_channels
        for rate, kernel in zip(upsample_rates, upsample_kernels):
            ch //= 2
            self.ups.append(tf.keras.layers.Conv1DTranspose(
                ch, kernel, strides=rate, padding='same'
            ))
            for k, d in zip(resblock_kernels, resblock_dilations):
                self.resblocks.append(ResBlock(ch, k, d))

        self.post_conv = tf.keras.layers.Conv1D(1, 7, padding='same', activation='tanh')

    def call(self, mel, training=False):
        # mel: [B, T_mel, mel_channels]
        x = self.pre_conv(mel)
        rb_idx = 0
        for up in self.ups:
            x = tf.nn.leaky_relu(x, 0.1)
            x = up(x)
            # Suma de resblocks paralelos
            xs = sum(self.resblocks[rb_idx + j](x) for j in range(3))
            x = xs / 3
            rb_idx += 3
        x = tf.nn.leaky_relu(x, 0.1)
        return self.post_conv(x)  # [B, T_wav, 1]
M9 · Separación
Separación de Fuentes — SI-SDR y Máscaras
Scale-Invariant SDR, Ideal Ratio Mask, Conv-TasNet en TF
SI-SDR = 10·log₁₀(||αs||² / ||αs − ŝ||²)
donde α = ⟨ŝ,s⟩/||s||² (proyección de ŝ sobre s)
IRM: M(f,t) = |S(f,t)|² / (|S(f,t)|² + |N(f,t)|²)

SI-SDR Loss

import tensorflow as tf

def si_sdr_loss(target: tf.Tensor, estimate: tf.Tensor,
                eps: float = 1e-8) -> tf.Tensor:
    """
    Scale-Invariant SDR — métrica estándar para separación de fuentes.
    Invariante a la escala: mide solo la forma, no la amplitud.
    target, estimate: [batch, samples]
    Retorna: loss escalar (negativo del SI-SDR, para minimizar)
    """
    # Restar media (zero-mean)
    target   = target   - tf.reduce_mean(target,   axis=-1, keepdims=True)
    estimate = estimate - tf.reduce_mean(estimate, axis=-1, keepdims=True)

    # Proyección
    dot = tf.reduce_sum(target * estimate, axis=-1, keepdims=True)
    s_norm_sq = tf.reduce_sum(tf.square(target), axis=-1, keepdims=True) + eps
    alpha = dot / s_norm_sq

    s_target = alpha * target
    e_noise = estimate - s_target

    si_sdr = 10 * tf.math.log(
        tf.reduce_sum(tf.square(s_target), axis=-1) /
        (tf.reduce_sum(tf.square(e_noise), axis=-1) + eps) + eps
    ) / tf.math.log(10.0)

    return -tf.reduce_mean(si_sdr)  # negativo para minimizar

# Conv-TasNet simplificado para separación en TF
class ConvTasNetEncoder(tf.keras.layers.Layer):
    """Encoder: mapea waveform a representación latente."""
    def __init__(self, N=512, L=16):
        super().__init__()
        self.conv = tf.keras.layers.Conv1D(N, L, strides=L//2, use_bias=False)
    def call(self, x):

class AudioClassifier(tf.keras.Model):
    def __init__(self, num_classes: int):
        super().__init__()
        self.yamnet = yamnet_model
        # Congelar YAMNet durante las primeras épocas
        self.dense1 = tf.keras.layers.Dense(256, activation='relu')
        self.dropout = tf.keras.layers.Dropout(0.3)
        self.head = tf.keras.layers.Dense(num_classes, activation='softmax')

    def call(self, waveform, training=False):
        _, embeddings, _ = self.yamnet(waveform)
        x = tf.reduce_mean(embeddings, axis=0, keepdims=True)  # mean pooling
        x = self.dense1(x)
        x = self.dropout(x, training=training)
        return self.head(x)

import tensorflow as tf

# Connectionist Temporal Classification (CTC)
# Permite entrenar sin alineación explícita audio↔texto
# La red aprende el alignment implícitamente

# Input: log-mel spectrogram [batch, time, mel_bins]
# Output: distribución sobre caracteres + blank token

class CTCModel(tf.keras.Model):
    def __init__(self, num_chars: int, mel_bins: int = 128):
        super().__init__()
        # Encoder convolucional
        self.conv1 = tf.keras.layers.Conv2D(32, (3,3), activation='relu', padding='same')
        self.pool1 = tf.keras.layers.MaxPool2D((1,2))
        self.conv2 = tf.keras.layers.Conv2D(64, (3,3), activation='relu', padding='same')
        self.pool2 = tf.keras.layers.MaxPool2D((1,2))
        # RNN bidireccional
        self.birnn = tf.keras.layers.Bidirectional(
            tf.keras.layers.LSTM(256, return_sequences=True)
        )
        # Proyección a vocabulario
        self.fc = tf.keras.layers.Dense(num_chars + 1)  # +1 para blank

    def call(self, x, training=False):
        # x: [B, T, F] → [B, T, F, 1]
        x = x[..., tf.newaxis]
        x = self.pool1(self.conv1(x))
        x = self.pool2(self.conv2(x))
        # Reshape: [B, T, F//4 * 64]
        B, T = tf.shape(x)[0], tf.shape(x)[1]
        x = tf.reshape(x, [B, T, -1])
        x = self.birnn(x, training=training)
        return self.fc(x)  # [B, T, vocab+1] logits

@tf.function
def ctc_train_step(model, optimizer, x_batch, labels, label_lengths, input_lengths):
    with tf.GradientTape() as tape:
        logits = model(x_batch, training=True)
        log_probs = tf.nn.log_softmax(logits)
        # CTC Loss (tensorflow incluye implementación nativa)
        loss = tf.nn.ctc_loss(
            labels=tf.cast(labels, tf.int32),
            logits=tf.transpose(log_probs, [1,0,2]),  # [T,B,V]
            label_length=label_lengths,
            logit_length=input_lengths,
            blank_index=0
        )
    grads = tape.gradient(loss, model.trainable_variables)
    optimizer.apply_gradients(zip(grads, model.trainable_variables))
    return tf.reduce_mean(loss)

SAP Loss — Sentence-Audio Prediction

import tensorflow as tf

def contrastive_audio_text_loss(audio_emb: tf.Tensor,
                                 text_emb: tf.Tensor,
                                 temperature: float = 0.07) -> tf.Tensor:
    """
    Pérdida contrastiva InfoNCE para pares (audio, texto).
    Similar a CLIP pero para audio + transcripciones / descripciones.

    audio_emb: [B, d_audio] — embeddings de audio
    text_emb:  [B, d_text]  — embeddings de texto (BERT/T5)
    """
    # Proyectar a espacio común
    # (asumimos ya proyectados a misma dimensión d)

    # Normalizar a esfera unitaria
    audio_emb = tf.math.l2_normalize(audio_emb, axis=-1)
    text_emb  = tf.math.l2_normalize(text_emb,  axis=-1)

    # Matriz de similitud [B, B]
    logits = tf.matmul(audio_emb, text_emb, transpose_b=True) / temperature

    # Labels: diagonal (pares positivos)
    B = tf.shape(logits)[0]
    labels = tf.range(B)

    # Loss simétrica: audio→texto + texto→audio
    loss_a2t = tf.keras.losses.sparse_categorical_crossentropy(
        labels, logits, from_logits=True
    )
    loss_t2a = tf.keras.losses.sparse_categorical_crossentropy(
        labels, tf.transpose(logits), from_logits=True
    )
    return (tf.reduce_mean(loss_a2t) + tf.reduce_mean(loss_t2a)) / 2

Examen Feynman

Q1 · STFT en el grafo

¿Por qué es importante que tf.signal.stft sea diferenciable? ¿Qué permite hacer esto en el entrenamiento de modelos generativos como DDSP?

Ver respuesta

Si la STFT es diferenciable, podemos calcular gradientes ∂L/∂x donde L es la MR-STFT Loss en el dominio espectral y x es la forma de onda generada. Esto permite entrenar DDSP y HiFi-GAN con la pérdida directamente sobre el espectrograma sin necesidad de una discriminador adversarial, lo que hace el entrenamiento más estable. En tf.signal, todas las operaciones (ventana Hann, FFT, módulo) son diferenciables, por lo que el gradiente fluye desde el espectrograma hasta los parámetros del sintetizador.

Q2 · STE en QAT

¿Por qué la función round() no se puede usar directamente en backpropagation y cómo el Straight-Through Estimator resuelve esto?

Ver respuesta

round(x) tiene derivada 0 en casi todos los puntos (es una función escalonada). Si se usara en backprop, los gradientes serían 0 en todas partes → el modelo no aprendería nada. El STE reemplaza ∂round(x)/∂x por 1 (identidad) durante backprop: los gradientes pasan directamente como si round() no existiera. En forward, se hace la cuantización real (simula el error int8). En backward, gradiente fluye como si fuera un nodo de identidad. Matemáticamente es incorrecto pero empíricamente funciona muy bien para entrenar redes cuantizadas.

Q3 · MR-STFT Loss

¿Por qué usar múltiples tamaños de FFT en MR-STFT Loss en lugar de un solo tamaño grande?

Ver respuesta

Un FFT grande (2048) tiene buena resolución en frecuencia pero mala resolución temporal → no puede capturar ataques rápidos (<1ms). Un FFT pequeño (256) tiene buena resolución temporal pero bandas de frecuencia anchas → no distingue armónicos cercanos. Con múltiples resoluciones, la pérdida penaliza errores tanto en la estructura de pitch (FFT largo) como en la envolvente temporal (FFT corto). Esto produce vocoder generados con calidad perceptual superior en todo el espectro.

Q4 · Mixed Precision

¿Por qué se acumula en FP32 incluso si los cálculos se hacen en FP16? ¿Qué pasaría si se acumulara en FP16?

Ver respuesta

FP16 tiene precisión de ~3 dígitos decimales y rango hasta 65504. En una suma de N términos, el error de redondeo crece como O(N·ε_mach). En una matmul de matrices 512×512, hay 512 sumas parciales → el error acumulado en FP16 puede ser significativo, causando divergencia del entrenamiento o pérdida de precisión. FP32 tiene precisión de ~7 dígitos, suficiente para acumular sumas largas. Los Tensor Cores hacen la multiplicación en FP16 pero acumulan internamente en FP32 antes de escribir el resultado.

Q5 · RaggedTensor

¿Qué estructura de datos usa tf.RaggedTensor internamente para representar tensores de longitud variable sin desperdicio de memoria?

Ver respuesta

Un RaggedTensor tiene dos componentes: flat_values (un tensor 1D denso que contiene todas las muestras concatenadas sin padding) y row_splits (un tensor de índices enteros que marca dónde empieza y termina cada "fila"). Para 3 audios de 10000, 8000 y 15000 muestras: flat_values tiene 33000 elementos, row_splits=[0, 10000, 18000, 33000]. Esto evita el padding que desperdiciaría hasta 2× la memoria en casos de longitud muy variable.
